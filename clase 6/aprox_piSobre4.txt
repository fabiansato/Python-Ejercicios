import math
n = int(input("Ingrese número entero")) #cuanto mayor el numero de term, mejor la aprox
# suma de los primeros n términos de la serie.
#alternada ejercicio 15 práctica 3
suma = 1
for i in range(1, n+1):
    suma = suma + 1/((-1)**i * (2*i + 1))

print ("La suma hasta el término n es ", suma)
