def coincide(palabra, palAux):
    for i in range(len(palAux)):
        if(palabra[i]!=palAux[i]):
            return False
    return True

def esPrimo(numero):
	
    for i in range(2, numero):
        if(numero%i==0):
            return False
    return True

def trePrimosMayores(num):
    cont=0
    lista=[]
    num+=1
    while(cont<3):
        if(esPrimo(num)):
            cont+=1
            lista.append(num)
        num+=1
    return(lista)
